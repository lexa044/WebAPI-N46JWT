	@inId int,

ALTER PROCEDURE [dbo].[API_UpdateCustomerByIdNumber]
	@inIdType smallint,
	@inIdNumber varchar(16)
AS

	SET NOCOUNT ON;

	UPDATE [dbo].[Customers]
	SET
		IsPositive = CASE WHEN IsPositive = 0 THEN 1 ELSE 0 END
	WHERE [IdType] = @inIdType AND [IdNumber]=@inIdNumber;

	SELECT 
	   [Id]
      ,[IdType]
      ,[IdNumber]
      ,[IsPositive]
  FROM [dbo].[Customers]
  WHERE [IdType] = @inIdType AND [IdNumber]=@inIdNumber;

  SET NOCOUNT OFF;

GO
